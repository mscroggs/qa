# The quadrature is not exact as integrand (in mathematical sense)
# is not polynomial - UFL, in fact, operates DOF-wise and keeps
# polynomial basis.
F = abs(u)*dx

# Customize quadrature
F = abs(u)*dx(None, {'quadrature_degre': 5})
