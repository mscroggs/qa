commented Jul 2, 2013 by johanhake FEniCS Expert (22,480 points)

Generate mesh once, store to file and reuse to ensure deterministic behaviour of your program.

from dolfin import *
try:
    mesh = Mesh('mesh.xml.gz')
except RuntimeError:
    c = Point(0, 0, 0)
    S = Sphere(c, 8)
    mesh = Mesh(S, 16)
    File('mesh.xml.gz') << mesh
print mesh
