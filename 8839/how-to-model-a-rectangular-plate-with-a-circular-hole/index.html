Use the mshr-package (https://bitbucket.org/fenics-project/mshr):

from dolfin import *    
import mshr
domain = mshr.Rectangle(Point(0,0), Point(4,1))-mshr.Circle(Point(2,0.5), 0.2)
mesh = mshr.generate_mesh(domain, 60)
plot(mesh, interactive=True)
