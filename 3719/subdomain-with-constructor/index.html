class OutFlow(SubDomain):

Try:

class OutFlow(SubDomain):
def __init__(self,y1,y2):
    self.y1 = y1
    self.y2 = y2
    SubDomain.__init__(self) # Call base class constructor!
def inside(self, x, on_boundary):
    return (near(x[0],0.0) and 
           between(x[1], (self.y1, self.y2)) and
           on_boundary)
