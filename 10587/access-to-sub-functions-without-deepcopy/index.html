0 votes

See the following example:

from dolfin import *

mesh = UnitSquareMesh(2, 2)
V = FunctionSpace(mesh, 'CG', 1)
W = MixedFunctionSpace([V, V])
w = Function(W)
v1, v2 = w.split(deepcopy=True)
print v1.vector().array()
